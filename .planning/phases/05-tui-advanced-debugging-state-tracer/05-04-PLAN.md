```rust
fn find_prev_event(&self, report: &DebugReport, current: usize, mode: SteppingMode) -> usize {
    if current == 0 {
        return 0;
    }

    match mode {
        SteppingMode::LineByLine => current.saturating_sub(1),
        SteppingMode::StateByState => {
            report.trace[0..current].iter().enumerate().rev()
                .find(|(_, e)| e.state_before != e.state_after)
                .map(|(i, _)| i)
                .unwrap_or(current)
        },
        SteppingMode::ActionByAction => {
            report.trace[0..current].iter().enumerate().rev()
                .find(|(_, e)| matches!(e.event_type, TraceEventType::RecordEmitted | TraceEventType::RecordCleared))
                .map(|(i, _)| i)
                .unwrap_or(current)
        }
    }
}
```

And `jump_to_previous_record`:

```rust
pub fn jump_to_previous_record(&mut self) {
    let Some(report) = &self.last_good else { return };
    if self.trace_index == 0 {
        return;
    }

    if let Some((idx, _)) = report.trace[0..self.trace_index].iter().enumerate().rev()
        .find(|(_, e)| matches!(e.event_type, TraceEventType::RecordEmitted))
    {
        self.trace_index = idx;
        self.sync_cursor_to_trace();
    }
}
```
